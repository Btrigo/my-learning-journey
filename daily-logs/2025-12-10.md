- ipconfig             # Show IP config for all adapters
- ipconfig /all        # Detailed IP info (DHCP, DNS, MAC)

- ipconfig /renew      # Request new DHCP lease
- ipconfig /flushdns   # Clear local DNS cache

- ping <host/IP>       # Test reachability and basic latency
- tracert <host>       # Trace route to destination, hop by hop
- pathping <host>      # Ping + tracert combined with per-hop stats

- nslookup <host>      # Query DNS for record info
- nslookup <host> <dns># Query a specific DNS server

- getmac               # Show MAC address(es) of local NICs
- arp -a               # Show ARP cache (IP ‚Üî MAC mappings)
- route print          # Show routing table

- netstat -an          # Show active connections + listening ports

- netstat -u           # show UDP connections 

- hostname             # Show local computer name

- net use              # List or map network drives
